CREATE TABLE  "ADMINISTRATION" 
   (	"ADMIN_NR" NUMBER, 
	"PRES_NAME" VARCHAR2(30), 
	"YEAR_INAUGURATED" NUMBER, 
	 CONSTRAINT "ADMINISTRATION_CON" UNIQUE ("ADMIN_NR", "PRES_NAME", "YEAR_INAUGURATED") ENABLE
   )
/
CREATE TABLE  "ADMIN_PR_VP" 
   (	"ADMIN_NR" NUMBER, 
	"PRES_NAME" VARCHAR2(30), 
	"VICE_PRES_NAME" VARCHAR2(30), 
	 CONSTRAINT "ADMIN_PR_VP_CON" UNIQUE ("ADMIN_NR", "PRES_NAME", "VICE_PRES_NAME") ENABLE
   )
/
CREATE TABLE  "ELECTION" 
   (	"ELECTION_YEAR" NUMBER, 
	"CANDIDATE" VARCHAR2(30), 
	"VOTES" NUMBER, 
	"WINNER_LOSER_INDIC" VARCHAR2(15), 
	 CONSTRAINT "ELECTION_CON" UNIQUE ("ELECTION_YEAR", "CANDIDATE") ENABLE
   )
/
CREATE TABLE  "PRESIDENT" 
   (	"PRES_NAME" VARCHAR2(30), 
	"BIRTH_YR" NUMBER, 
	"YRS_SERV" NUMBER, 
	"DEATH_AGE" NUMBER, 
	"PARTY" VARCHAR2(30), 
	"STATE_BORN" VARCHAR2(30), 
	 CONSTRAINT "PRESIDENT_PK" PRIMARY KEY ("PRES_NAME") ENABLE
   )
/
CREATE TABLE  "PRES_HOBBY" 
   (	"PRES_NAME" VARCHAR2(30), 
	"HOBBY" VARCHAR2(30), 
	 CONSTRAINT "PRES_HOBBY_CON" UNIQUE ("PRES_NAME", "HOBBY") ENABLE
   )
/
CREATE TABLE  "PRES_MARRIAGE" 
   (	"PRES_NAME" VARCHAR2(30), 
	"SPOUSE_NAME" VARCHAR2(30), 
	"PR_AGE" NUMBER, 
	"SP_AGE" NUMBER, 
	"NR_CHILDREN" NUMBER, 
	"MAR_YEAR" NUMBER, 
	 CONSTRAINT "PRES_MARRIAGE_CON" UNIQUE ("PRES_NAME", "SPOUSE_NAME") ENABLE
   )
/
CREATE TABLE  "STATE" 
   (	"STATE_NAME" VARCHAR2(30), 
	"ADMIN_ENTERED" NUMBER, 
	"YEAR_ENTERED" NUMBER, 
	 CONSTRAINT "STATE_PK" PRIMARY KEY ("STATE_NAME") ENABLE
   )
/
